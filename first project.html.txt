    <input id="thickness" type="number" placeholder="Slab Thickness (mm)">
    <input id="Ec" type="number" placeholder="Concrete Modulus Ec (MPa)">

    <button onclick="calculate()">Calculate</button>

<script>

const steelProfiles = {
    "IPE200": { As: 2620, Es: 210000, h: 200 },
    "IPE300": { As: 5380, Es: 210000, h: 300 },
    "HEA200": { As: 5890, Es: 210000, h: 190 },
    "HEA300": { As: 11200, Es: 210000, h: 290 }
};

let selectedSteel = null;

function updateSteel() {
    let selected = document.getElementById("steelSelect").value;

    if (selected === "") {
        document.getElementById("steelData").innerHTML = "";
        selectedSteel = null;
        return;
    }

    selectedSteel = steelProfiles[selected];

    document.getElementById("steelData").innerHTML =
        "<b>Steel Properties:</b><br>" +
        "Area (As): " + selectedSteel.As + " mm²<br>" +
        "Depth (h): " + selectedSteel.h + " mm<br>" +
        "Es: " + selectedSteel.Es + " MPa";
}

function calculate() {

    if (!selectedSteel) {
        document.getElementById("result").innerText = "Select a steel profile.";
        return;
    }

    let width = parseFloat(document.getElementById("width").value);
    let thickness = parseFloat(document.getElementById("thickness").value);
    let Ec = parseFloat(document.getElementById("Ec").value);

    if (isNaN(width) || isNaN(thickness) || isNaN(Ec)) {
        document.getElementById("result").innerText = "Enter slab details.";
        return;
    }

    let As = selectedSteel.As;
    let Es = selectedSteel.Es;
    let h = selectedSteel.h;

    let Ac = width * thickness;

    let n = Es / Ec;

    let Ac_trans = Ac * n;

    // centroid positions (from bottom of steel)
    let ys = h / 2;
    let yc = h + thickness / 2;

    let ybar = (As * ys + Ac_trans * yc) / (As + Ac_trans);

    document.getElementById("result").innerHTML =
        "<b>Results:</b><br>" +
        "Concrete Area: " + Ac.toFixed(0) + " mm²<br>" +
        "Modular Ratio (n): " + n.toFixed(2) + "<br>" +
        "Neutral Axis (ȳ): " + ybar.toFixed(2) + " mm from bottom<br>";
}

</script>
